class Window:

    def init(self):
        self.hideChatWindow()
        self.hideUsersBar()
        self.turnSoundsOff()
        self.goFullScreen()
        wait(0.5)
        # bug - opening majors house dialog
        dialog = exists()
        if dialog:
            click(dialog)


    def hideUsersBar(self):
        arrow = exists(Pattern().similar(0.90))
        if arrow:
            click(arrow)

    def hideChatWindow(self):
        button = exists(Pattern().similar(0.90).targetOffset(0,-13))
        if button:
            click(button)

    def turnSoundsOff(self):
        menu = exists(Pattern().similar(0.90))
        if (menu):
            click(menu)
        wait(1)
        note = exists(Pattern().similar(0.80))
        if (note):
            click(note)
        speaker = exists(Pattern().similar(0.90))
        if (speaker):
            click(speaker)
        arrow = find()
        click(arrow)

    def goFullScreen(self):
        if exists(Pattern().similar(0.90)):
            green = find()
            click(green)
            type(Key.F11)
            confirm = exists()
            if confirm:
                click(confirm)
